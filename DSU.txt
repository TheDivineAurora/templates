#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define ld long double
using pii = pair<ll,ll>;
const ll INF  = 1e9 + 5;
const int N = 5e5;

vector<int> parent;
vector<int> sz;

void make(int v){
  parent[v] = v;
  sz[v]=1;
}

int find(int v){
  if(parent[v]==v) return v;
  return parent[v] = find(parent[v]);
}

void Union(int a,int b){
  a = find(a);
  b = find(b);
  if(a!=b){
    if(sz[a]<sz[b]) 
       swap(a,b);
    parent[b] = a;
    sz[a]+=sz[b];
  }
}
void solve(){
int n,m;
cin>>n>>m;
parent.assign(n+1,0);
sz.assign(n+1,0);
for(int i = 1; i<=n; ++i){
  make(i);
}



}






  
int main(){
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
   ll t=1;
   for (ll i = 1; i <=t; ++i)
   {
      solve();
   }
}
